        async function fetchPrices() {

            if (!OANDA.connected) return null;

           

            try {

                const res = await fetch(

                    `${OANDA.baseUrl}/accounts/${OANDA.accountId}/pricing?instruments=${OANDA.pairs.join(',')}`,

                    { headers: { 'Authorization': `Bearer ${OANDA.apiKey}` } }

                );

               

                if (!res.ok) throw new Error('Price fetch failed');

               

                const data = await res.json();

                const prices = {};

               

                for (const p of data.prices) {

                    const name = p.instrument === 'XAU_USD' ? 'XAUUSD' : 'GBPJPY';

                    prices[name] = {

                        bid: parseFloat(p.bids[0].price),

                        ask: parseFloat(p.asks[0].price),

                        mid: (parseFloat(p.bids[0].price) + parseFloat(p.asks[0].price)) / 2,

                        spread: parseFloat(p.asks[0].price) - parseFloat(p.bids[0].price),

                        time: new Date(p.time)

                    };

                    STATE.prices[name] = prices[name];

                }

               

                return prices;

            } catch (e) {

                return null;

            }

        }

       

        async function fetchCandles(pair, count = 30) {

            if (!OANDA.connected) return null;

           

            const inst = pair === 'XAUUSD' ? 'XAU_USD' : 'GBP_JPY';

           

            try {

                const res = await fetch(

                    `${OANDA.baseUrl}/accounts/${OANDA.accountId}/instruments/${inst}/candles?` +

                    `granularity=M1&count=${count}&price=M`,

                    { headers: { 'Authorization': `Bearer ${OANDA.apiKey}` } }

                );

               

                if (!res.ok) throw new Error('Candle fetch failed');

               

                const data = await res.json();

                const candles = data.candles.map(c => ({

                    time: new Date(c.time),

                    open: parseFloat(c.mid.o),

                    low: parseFloat(c.mid.l),

                    close: parseFloat(c.mid.c),

                    volume: c.volume

                }));

               

                STATE.candles[pair] = candles;

                return candles;

            } catch (e) {

                return null;

            }

        }

        function calcEMA(data, period) {

            if (data.length < period) return null;

            const k = 2 / (period + 1);

            let ema = data[0].close;

            for (let i = 1; i < data.length; i++) {

                ema = data[i].close * k + ema * (1 - k);

            }

            return ema;

        }

       

        function calcHeikenAshi(candles) {

            if (candles.length < 2) return null;

            const ha = [];

           

            ha.push({

                close: (candles[0].open + candles[0].high + candles[0].low + candles[0].close) / 4,

                open: (candles[0].open + candles[0].close) / 2,

                high: candles[0].high,

                low: candles[0].low,

                color: candles[0].close >= candles[0].open ? 'green' : 'red'

            });

           

            for (let i = 1; i < candles.length; i++) {

                const haClose = (candles[i].open + candles[i].high + candles[i].low + candles[i].close) / 4;

                const haOpen = (ha[i-1].open + ha[i-1].close) / 2;

                ha.push({

                    close: haClose,

                    open: haOpen,

                    high: Math.max(candles[i].high, haOpen, haClose),

                    low: Math.min(candles[i].low, haOpen, haClose),

                    color: haClose >= haOpen ? 'green' : 'red'

                });

            }

           

            return ha;

        }

       

        function calcStochRSI(candles, rsiPeriod = 14, stochK = 5) {

            if (candles.length < rsiPeriod + stochK) return null;

           

            const closes = candles.map(c => c.close);

            let gains = 0, losses = 0;

           

            for (let i = closes.length - rsiPeriod; i < closes.length; i++) {

                const change = closes[i] - closes[i-1];

                if (change > 0) gains += change;

                else losses -= change;

            }

           

            const avgGain = gains / rsiPeriod;

            const avgLoss = losses / rsiPeriod;

            const rsi = avgLoss === 0 ? 100 : 100 - (100 / (1 + avgGain / avgLoss));

           

            const recentCloses = closes.slice(-stochK);

            const highest = Math.max(...recentCloses);

            const lowest = Math.min(...recentCloses);

            const stoch = highest === lowest ? 50 :

                ((closes[closes.length-1] - lowest) / (highest - lowest)) * 100;

           

            return stoch;

        }

       

        function analyzeICC(pair, candles) {

            const result = {

                pair, direction: null, confidence: 0, price: 0,

                conds: { ema: false, ha: false, stoch: false },

                details: {}

            };

           

            if (!candles || candles.length < 25) return result;

           

            const ema21 = calcEMA(candles, 21);

            const ha = calcHeikenAshi(candles);

            const stoch = calcStochRSI(candles);

            const prevCandles = candles.slice(0, -1);

            const prevStoch = calcStochRSI(prevCandles);

           

            if (!ema21 || !ha || stoch === null) return result;

           

            const price = candles[candles.length-1].close;

            result.price = price;

            result.details.ema21 = ema21;

            result.details.stoch = stoch;

            result.details.haColor = ha[ha.length-1].color;

            result.details.prevHaColor = ha.length > 1 ? ha[ha.length-2].color : null;

           

            result.conds.ema = Math.abs(price - ema21) / ema21 < 0.0015;

           

            if (ha.length >= 2) {

                result.conds.ha = ha[ha.length-1].color !== ha[ha.length-2].color;

            }

           

            if (stoch < 20 && prevStoch !== null && stoch > prevStoch) {

                result.conds.stoch = 'oversold_rising';

            } else if (stoch > 80 && prevStoch !== null && stoch < prevStoch) {

                result.conds.stoch = 'overbought_falling';

            }

           

            if (result.conds.ema && result.conds.ha) {

                if (result.details.haColor === 'green' && result.conds.stoch === 'oversold_rising') {

                    result.direction = 'BUY';

                } else if (result.details.haColor === 'red' && result.conds.stoch === 'overbought_falling') {

                    result.direction = 'SELL';

                }

            }

           

            if (result.direction) {

                let score = 30 + 35 + 35;

                if (result.conds.stoch === 'oversold_rising' && stoch < 15) score += 10;

                if (result.conds.stoch === 'overbought_falling' && stoch > 85) score += 10;

                result.confidence = Math.min(score, 100);

            }

           

            return result;

        }

        async function scan() {

            if (!STATE.running || !OANDA.connected) return;

           

            STATE.scans++;

            document.getElementById('scanBadge').textContent = STATE.scans + ' scans';

           

            for (const oandaPair of OANDA.pairs) {

                const pair = oandaPair === 'XAU_USD' ? 'XAUUSD' : 'GBPJPY';

               

                try {

                    const candles = await fetchCandles(pair, 30);

                    const prices = await fetchPrices();

                   

                    if (candles && prices && prices[pair]) {

                        const bidEl = document.getElementById(pair.toLowerCase() + 'Bid');

                        if (bidEl) {

                            bidEl.textContent = prices[pair].bid.toFixed(pair === 'XAUUSD' ? 2 : 3);

                        }

                       

                        const signal = analyzeICC(pair, candles);

                        updateSignalUI(pair, signal);

                       

                        if (signal.direction && signal.confidence >= 70) {

                            fireAlert(signal);

                        }

                    }

                } catch (e) {

                    // continue

                }

            }

        }

        function updateSignalUI(pair, signal) {

            const dirEl = document.getElementById(pair.toLowerCase() + 'Dir');

        function fireAlert(signal) {

            const overlay = document.getElementById('alertOverlay');

            const card = document.getElementById('alertCard');

            const symbol = document.getElementById('alertSymbol');

            const pairEl = document.getElementById('alertPair');

            const signalEl = document.getElementById('alertSignal');

            const priceEl = document.getElementById('alertPrice');

           

            card.className = 'alert-card ' + signal.direction.toLowerCase();

            symbol.textContent = signal.direction === 'BUY' ? '📈' : '📉';

            pairEl.textContent = signal.pair;

            signalEl.textContent = `${signal.direction} SIGNAL • ${signal.confidence}%`;

            priceEl.textContent = `Entry: ${signal.price.toFixed(signal.pair==='XAUUSD'?2:3)} | EMA: ${signal.details.ema21?.toFixed(2)}`;

           

            overlay.classList.add('active');

           

            if (navigator.vibrate) {

                navigator.vibrate([200, 100, 200, 100, 400]);

            }

           

            playBeep();

           

            if ('Notification' in window && Notification.permission === 'granted') {

                try {

                    navigator.serviceWorker.ready.then(reg => {

                        reg.showNotification(`ICC ${signal.direction}: ${signal.pair}`, {

                            body: `Entry ${signal.price.toFixed(2)} • ${signal.confidence}% confidence`,

                            icon: '📈',

                            vibrate: [200, 100, 200, 100, 400],

                            tag: 'icc-alert',

                            requireInteraction: true,

                        });

                    });

                } catch(e) {

                    new Notification(`ICC ${signal.direction}: ${signal.pair}`, {

                        body: `Entry ${signal.price.toFixed(2)}`,

                    });

                }

            }

           

            STATE.alerts++;

            addLog(`🚨 ${signal.pair} ${signal.direction} @ ${signal.price.toFixed(2)} [${signal.confidence}%]`, 'alert');

           

            setTimeout(() => {

                overlay.classList.remove('active');

            }, 45000);

        }

       

        function dismissAlert() {

        function playBeep() {

            try {

                const ctx = new (window.AudioContext || window.webkitAudioContext)();

                [800, 1000, 1200].forEach((f, i) => {

                    const osc = ctx.createOscillator();

                    const gain = ctx.createGain();

                    osc.connect(gain); gain.connect(ctx.destination);

                    osc.type = 'square';

                    osc.frequency.value = f;

                    gain.gain.setValueAtTime(0.3, ctx.currentTime + i*0.12);

                    gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + i*0.12+0.1);

                    osc.start(ctx.currentTime + i*0.12);

                    osc.stop(ctx.currentTime + i*0.12+0.1);

                });

            } catch(e) {}

        }

        function startBot() {

            if (!OANDA.connected) {

                showToast('Connect API first!');

                return;

            }

            if (STATE.running) return;

           

            STATE.running = true;

            document.getElementById('btnStart').disabled = true;

            document.getElementById('btnStop').disabled = false;

            document.getElementById('statusBadge').textContent = '● LIVE';

           

            if ('Notification' in window && Notification.permission === 'default') {

                Notification.requestPermission();

            }

           

            addLog('🤖 Bot STARTED • 1m ICC Strategy', 'ok');

            scan();

            STATE.timer = setInterval(scan, OANDA.pollMs);

            showToast('Monitoring started 📊');

        }

       

        function stopBot() {

            STATE.running = false;

            document.getElementById('btnStart').disabled = false;

            document.getElementById('btnStop').disabled = true;

            document.getElementById('statusBadge').textContent = '● ONLINE';

           

            if (STATE.timer) {

                clearInterval(STATE.timer);

                STATE.timer = null;

            }

           

            addLog('⏹ Bot stopped', 'info');

            showToast('Monitoring stopped');

        }

        function addLog(msg, type = 'info') {

            const container = document.getElementById('logContent');

            const line = document.createElement('div');

            line.className = 'log-line ' + (type === 'alert' ? 'alert' : type === 'error' ? 'error' : type === 'ok' ? 'ok' : '');

            line.textContent = `[${new Date().toLocaleTimeString()}] ${msg}`;

            container.insertBefore(line, container.firstChild);

            while (container.children.length > 80) container.lastChild.remove();

        }
